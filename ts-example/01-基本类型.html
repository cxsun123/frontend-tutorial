01-基本类型 .ts

字符串

使用 string 类型声明字符串变量

let greeting: string = "Hello, TypeScript!";
console.log(greeting);

数字

使用 number 类型声明数字，支持整数和浮点数

let age: number = 25;
let price: number = 99.9;
console.log(`Age: ${age}, Price: ${price}`);

布尔值

使用 boolean 类型声明 true/false 值

let isDone: boolean = false;
console.log(`Is done: ${isDone}`);

null 和 undefined

JavaScript 的两个特殊值

let nothing: null = null;
let notDefined: undefined = undefined;
console.log(nothing, notDefined);

数组

两种声明方式：number[] 或 Array

let numbers: number[] = [1, 2, 3, 4, 5];
let names: Array<string> = ["Alice", "Bob", "Charlie"];
console.log(numbers, names);

元组 (Tuple)

固定长度和类型的数组 [string, number]

let tuple: [string, number] = ["Alice", 30];
console.log(tuple);

枚举 (Enum)

一组命名常量，可用数字或字符串值

enum Direction {
  Up,
  Down,
  Left,
  Right,
}
let dir: Direction = Direction.Up;
console.log(`Direction: ${Direction[dir]}`); // "Up"

enum Color {
  Red = "RED",
  Green = "GREEN",
  Blue = "BLUE",
}
let favoriteColor: Color = Color.Red;
console.log(`Color: ${favoriteColor}`);

任意值 (any)

关闭类型检查，可赋任意值（不推荐）

let anything: any = 42;
anything = "now a string";
anything = true;
console.log(anything);

void

没有返回值的函数

function sayHello(): void {
  console.log("Hello!");
}
sayHello();

never

永远不会返回的函数（抛出异常或死循环）

function throwError(msg: string): never {
  throw new Error(msg);
}
// throwError("oops"); // 取消注释会抛出错误
