09 - Next.js 数据获取与 API

全栈数据获取方案

Next.js 提供多种数据获取方式，并内置 API Routes 让你在同一个项目中编写后端接口。

服务端组件获取数据

在 App Router 中，组件默认是服务端组件，可直接 async

// app/posts/page.jsx — 服务端组件，直接 fetch
async function getPosts() {
  const res = await fetch('https://api.example.com/posts');
  // 默认缓存（SSG），设置 cache 控制策略
  return res.json();
}

export default async function PostsPage() {
  const posts = await getPosts();

  return (
    <div>
      <h1>文章列表</h1>
      {posts.map(post =&gt; (
        <article key={post.id}>
          <h2>{post.title}</h2>
          <p>{post.body}</p>
        </article>
      ))}
    </div>
  );
}
服务端组件中可以直接使用 async/await，无需 useState/useEffect。

缓存与重新验证

fetch 的缓存策略控制

// 1. 静态数据（SSG）— 构建时获取，永不重新获取
fetch(url);

// 2. 按需重新验证（ISR）
fetch(url, { next: { revalidate: 3600 } });
// 或使用单独的函数
export const revalidate = 3600; // 每 3600 秒重新生成

// 3. 动态数据（SSR）— 每次请求都获取
fetch(url, { cache: 'no-store' });

// 4. 手动重新验证
import { revalidatePath } from 'next/cache';
revalidatePath('/posts'); // 重新验证 /posts 页面
revalidateTag('posts'); // 重新验证标记为 'posts' 的 fetch

动态渲染 vs 静态生成

控制页面的渲染策略

// app/dashboard/page.jsx

// 动态渲染（每次请求重新生成）
export const dynamic = 'force-dynamic';

// 静态生成（默认）
export const dynamic = 'force-static';

// 增量静态生成
export const revalidate = 60; // 60 秒后重新生成

// 使用 cookies/headers 自动变为动态
import { cookies } from 'next/headers';

export default async function Page() {
  const cookieStore = cookies();
  const theme = cookieStore.get('theme');
  return <h1>主题: {theme?.value}</h1>;
}

API Routes

在 app/api/ 目录下创建 route.js 编写后端 API

// app/api/posts/route.js — GET /api/posts

const posts = [
  { id: 1, title: 'Hello World' },
  { id: 2, title: 'Next.js 教程' },
];

export async function GET() {
  return Response.json(posts);
}

export async function POST(request) {
  const body = await request.json();
  const newPost = { id: posts.length + 1, ...body };
  posts.push(newPost);
  return Response.json(newPost, { status: 201 });
}

// app/api/posts/[id]/route.js — GET /api/posts/1

export async function GET(_, { params }) {
  const post = posts.find(p => p.id === Number(params.id));
  if (!post) {
    return Response.json({ error: '未找到' }, { status: 404 });
  }
  return Response.json(post);
}

完整的 CRUD 示例

一个简单的 Todo API

// app/api/todos/route.js
let todos = [
  { id: 1, text: '学习 Next.js', done: false },
  { id: 2, text: '写项目', done: true },
];

export async function GET() {
  return Response.json(todos);
}

export async function POST(request) {
  const { text } = await request.json();
  const todo = { id: Date.now(), text, done: false };
  todos.push(todo);
  return Response.json(todo, { status: 201 });
}

// app/api/todos/[id]/route.js
export async function PUT(request, { params }) {
  const { done } = await request.json();
  const todo = todos.find(t => t.id === Number(params.id));
  if (todo) todo.done = done;
  return Response.json(todo);
}

export async function DELETE(_, { params }) {
  todos = todos.filter(t => t.id !== Number(params.id));
  return Response.json({ success: true });
}

客户端数据获取

在客户端组件中调用自己的 API

'use client';

import { useState, useEffect } from 'react';

export default function TodoApp() {
  const [todos, setTodos] = useState([]);

  useEffect(() => {
    fetch('/api/todos')
      .then(res => res.json())
      .then(setTodos);
  }, []);

  async function addTodo(text) {
    const res = await fetch('/api/todos', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ text }),
    });
    const newTodo = await res.json();
    setTodos(prev => [...prev, newTodo]);
  }

  return (
    <div>
      <button onClick={() => addTodo('新任务')}>添加</button>
      <ul>
        {todos.map(todo => (
          <li key={todo.id}>
            {todo.text} {todo.done ? '✅' : '❌'}
          </li>
        ))}
      </ul>
    </div>
  );
}

小结

  • 服务端组件可以直接 async/await 获取数据
  • fetch 的 cache 和 revalidate 控制缓存策略
  • API Routes 在 app/api/ 目录下编写后端代码
  • 支持 GET/POST/PUT/DELETE 等 HTTP 方法
  • 客户端组件通过 fetch 调用自己的 API 层
