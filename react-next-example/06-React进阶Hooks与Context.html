useRef — 引用 DOM

useRef 创建一个可变的引用对象，不会触发重新渲染

import { useRef, useEffect } from 'react';

function AutoFocus() {
  const inputRef = useRef(null);

  useEffect(() => {
    // 组件挂载后自动聚焦输入框
    inputRef.current.focus();
  }, []);

  return <input ref={inputRef} placeholder="自动聚焦" />;
}

// 存储可变值（不触发重渲染）
function Stopwatch() {
  const [count, setCount] = useState(0);
  const intervalRef = useRef(null);

  function start() {
    intervalRef.current = setInterval(() => {
      setCount(c => c + 1);
    }, 1000);
  }

  function stop() {
    clearInterval(intervalRef.current);
  }

  return (
    <div>
      <p>{count}s</p>
      <button onClick={start}>开始</button>
      <button onClick={stop}>停止</button>
    </div>
  );
}

useMemo — 性能优化

缓存计算结果，仅在依赖变化时重新计算

import { useMemo, useState } from 'react';

function ExpensiveList({ numbers, filter }) {
  // 仅当 numbers 或 filter 变化时重新计算
  const filtered = useMemo(() => {
    console.log('执行过滤计算...');
    return numbers.filter(n => n > filter);
  }, [numbers, filter]);

  return (
    <ul>
      {filtered.map((n, i) => (
        <li key={i}>{n}</li>
      ))}
    </ul>
  );
}

// useCallback — 缓存函数引用
import { useCallback } from 'react';

function Parent() {
  const [count, setCount] = useState(0);

  const increment = useCallback(() => {
    setCount(c => c + 1);
  }, []); // 函数引用永远不变

  return <Child onIncrement={increment} />;
}

useContext — 全局状态

Context API 无需 prop drilling，直接在任意层级获取数据

import { createContext, useContext, useState } from 'react';

// 1. 创建 Context
const ThemeContext = createContext('light');
const UserContext = createContext(null);

// 2. Provider 提供数据
function App() {
  const [theme, setTheme] = useState('light');
  const user = { name: '小明', role: 'admin' };

  return (
    <ThemeContext.Provider value={{ theme, setTheme }}>
      <UserContext.Provider value={user}>
        <Main />
      </UserContext.Provider>
    </ThemeContext.Provider>
  );
}

// 3. 任意子组件消费
function Navbar() {
  const { theme, setTheme } = useContext(ThemeContext);
  const user = useContext(UserContext);

  return (
    <nav style={{
      background: theme === 'dark' ? '#333' : '#fff'
    }}>
      <span>{user.name}</span>
      <button onClick={() =>
        setTheme(t => t === 'dark' ? 'light' : 'dark')
      }>
        切换主题
      </button>
    </nav>
  );
}

自定义 Hook

将组件逻辑提取为可复用的函数

// 自定义 Hook（必须以 use 开头）
function useWindowSize() {
  const [size, setSize] = useState({
    width: window.innerWidth,
    height: window.innerHeight
  });

  useEffect(() => {
    const handleResize = () => {
      setSize({
        width: window.innerWidth,
        height: window.innerHeight
      });
    };
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return size;
}

function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);

    fetch(url)
      .then(res => {
        if (!res.ok) throw new Error('请求失败');
        return res.json();
      })
      .then(data => {
        if (!cancelled) { setData(data); setError(null); }
      })
      .catch(err => {
        if (!cancelled) setError(err.message);
      })
      .finally(() => {
        if (!cancelled) setLoading(false);
      });

    return () => { cancelled = true; };
  }, [url]);

  return { data, loading, error };
}

// 使用自定义 Hook
function Page() {
  const { width } = useWindowSize();
  const { data, loading } = useFetch('/api/users');

  return (
    <div>
      <p>当前窗口宽度: {width}</p>
      {loading ? <p>加载中...</p> : <p>用户数: {data?.length}</p>}
    </div>
  );
}
