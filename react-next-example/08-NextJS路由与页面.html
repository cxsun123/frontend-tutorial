08 - Next.js 路由与页面

基于文件系统的路由

Next.js 的 App Router 使用文件系统定义路由，目录嵌套即路由嵌套。

基本路由

文件路径对应 URL 路径

// app/page.jsx           →  /（首页）
// app/about/page.jsx     →  /about
// app/blog/page.jsx      →  /blog
// app/contact/page.jsx   →  /contact

// 每个 page.jsx 导出默认组件
export default function AboutPage() {
  return (
    <div>
      <h1>关于我们</h1>
      <p>这是一个 Next.js 示例应用</p>
    </div>
  );
}

动态路由

用方括号 [param] 创建动态路由

// app/blog/[slug]/page.jsx  →  /blog/hello-world

export default function BlogPost({ params }) {
  return <h1>文章: {params.slug}</h1>;
}

// app/products/[category]/[id]/page.jsx
// → /products/electronics/123

export default function ProductPage({ params }) {
  return (
    <div>
      <p>分类: {params.category}</p>
      <p>商品 ID: {params.id}</p>
    </div>
  );
}

// app/[...slug]/page.jsx    → 捕获所有路径
// app/[[...slug]]/page.jsx  → 可选捕获所有路径

导航组件 — Link

next/link 提供客户端导航，无页面刷新

import Link from 'next/link';

export default function Navbar() {
  return (
    <nav>
      <Link href="/">首页</Link>
      <Link href="/about">关于</Link>
      <Link
        href="/blog/post-1"
        className="nav-link"
      >
        博客文章
      </Link>

      {/* 动态路由 */}
      <Link href={`/products/${category}/${id}`}>
        商品详情
      </Link>

      {/* 替换历史记录 */}
      <Link href="/dashboard" replace>
        仪表盘
      </Link>
    </nav>
  );
}
Link 默认预取视口中的链接页面，实现即时导航。

编程式导航 — useRouter

用 useRouter Hook 在代码中跳转

'use client'; // 需要在客户端组件中使用

import { useRouter } from 'next/navigation';

export default function LoginButton() {
  const router = useRouter();

  function handleLogin() {
    // 登录逻辑...
    router.push('/dashboard');       // 跳转
    router.replace('/dashboard');    // 替换（不能后退）
    router.back();                   // 后退
    router.refresh();                // 刷新当前页
  }

  return <button onClick={handleLogin}>登录</button>;
}

嵌套布局

Layout 可以嵌套，实现不同区域的共享 UI

// app/dashboard/layout.jsx
export default function DashboardLayout({ children }) {
  return (
    <section>
      <aside>
        <nav>侧边栏导航...</nav>
      </aside>
      <main>{children}</main>
    </section>
  );
}

// app/dashboard/settings/page.jsx
export default function SettingsPage() {
  return <h1>设置页面</h1>;
}

// 结构:
// app/layout.jsx         → 全局（导航 + footer）
// app/dashboard/layout.jsx → 侧边栏（嵌套在全局内）
// app/dashboard/settings/page.jsx → 嵌套在侧边栏内

加载与错误处理

内置的 loading 和 error 组件

// app/blog/loading.jsx — 路由切换时显示
export default function Loading() {
  return (
    <div className="spinner">
      <p>加载中...</p>
    </div>
  );
}

// app/blog/error.jsx — 运行时错误的兜底 UI
'use client';

export default function Error({ error, reset }) {
  return (
    <div>
      <h2>出错了!</h2>
      <p>{error.message}</p>
      <button onClick={() => reset()}>重试</button>
    </div>
  );
}

小结

  • page.jsx 定义页面，文件路径即路由路径
  • [param] 动态路由匹配 URL 参数
  • Link 组件实现客户端导航，useRouter 实现编程式导航
  • Layout 可以嵌套，实现多层级共享布局
  • loading.jsx 和 error.jsx 实现加载态和错误处理
