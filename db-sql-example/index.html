快速开始

# 安装 PostgreSQL（Docker）
docker run --name pg -e POSTGRES_PASSWORD=mysecret -d -p 5432:5432 postgres:16

# 安装 Python 驱动
pip install psycopg2-binary

# 连接数据库
psql -U postgres -h localhost
