02 - SQL 基础

SQL 基础 — 数据库操作的核心

本章涵盖 SELECT、INSERT、UPDATE、DELETE 四大 DML 语句，以及 WHERE、ORDER BY、LIMIT 等常用子句。

创建表示例

本章示例基于以下 Product（商品）表

-- 创建数据库
CREATE DATABASE shop;

-- 创建商品表
CREATE TABLE Product (
    product_id   CHAR(4)      NOT NULL PRIMARY KEY,
    product_name VARCHAR(100) NOT NULL,
    price        INTEGER      NOT NULL,
    category     VARCHAR(50),
    regist_date  DATE
);

-- 插入示例数据

INSERT INTO Product VALUES ('0004', '菜刀', 1500, '厨房', '2024-04-05');
INSERT INTO Product VALUES ('0005', '高压锅', 6800, '厨房', '2024-05-01');

SELECT — 查询数据

最基本的查询语句

-- 查询所有列
SELECT * FROM Product;

-- 查询指定列
SELECT product_name, price FROM Product;

-- 为列起别名 (AS)
SELECT product_name AS 名称, price AS 价格 FROM Product;

-- 去重查询 (DISTINCT)
SELECT DISTINCT category FROM Product;

-- 常数列
SELECT '商品信息' AS 标题, product_name FROM Product;

WHERE — 条件筛选

过滤出符合条件的行

-- 比较运算符
SELECT * FROM Product WHERE price > 1000;
SELECT * FROM Product WHERE category = '衣物';
SELECT * FROM Product WHERE price BETWEEN 500 AND 3000;

-- 逻辑运算符
SELECT * FROM Product
WHERE category = '衣物' AND price >= 1000;

SELECT * FROM Product
WHERE category = '厨房' OR price >= 5000;

SELECT * FROM Product
WHERE NOT category = '办公';

-- IN 和 NOT IN
SELECT * FROM Product
WHERE category IN ('衣物', '厨房');

-- LIKE 模糊匹配
SELECT * FROM Product
WHERE product_name LIKE '%T%';

-- IS NULL 判断空值
SELECT * FROM Product WHERE regist_date IS NULL;

ORDER BY — 排序

对查询结果进行排序

-- 升序排列 (ASC)
SELECT product_name, price
FROM Product
ORDER BY price ASC;

-- 降序排列 (DESC)
SELECT product_name, price
FROM Product
ORDER BY price DESC;

-- 多列排序
SELECT product_name, category, price
FROM Product
ORDER BY category ASC, price DESC;

LIMIT / OFFSET — 分页

限制返回的行数（不同数据库语法略有不同）

-- 只取前 3 行 (PostgreSQL / MySQL)
SELECT * FROM Product LIMIT 3;

-- 跳过前 2 行，取 3 行
SELECT * FROM Product LIMIT 3 OFFSET 2;

-- MySQL 另一种写法
SELECT * FROM Product LIMIT 2, 3;
SQL Server 使用 TOP 3，Oracle 使用 FETCH FIRST 3 ROWS ONLY。

INSERT — 插入数据

向表中添加新行

-- 全列插入（按表定义顺序）
INSERT INTO Product
VALUES ('0006', '叉子', 300, '厨房', '2024-06-01');

-- 指定列插入
INSERT INTO Product (product_id, product_name, price)
VALUES ('0007', '抹布', 200);

-- 多行插入
INSERT INTO Product VALUES
    ('0008', '杯子', 800, '厨房', '2024-06-10'),
    ('0009', '笔记本', 700, '办公', '2024-06-15');

UPDATE — 更新数据

修改表中已有的数据

-- 更新单列
UPDATE Product
SET price = 1200
WHERE product_id = '0001';

-- 更新多列
UPDATE Product
SET price = price * 1.1, regist_date = '2024-12-31'
WHERE category = '衣物';

-- ⚠️ 没有 WHERE 会更新所有行
UPDATE Product SET price = 0;

DELETE — 删除数据

从表中删除行

-- 删除指定行
DELETE FROM Product
WHERE product_id = '0009';

-- 删除所有行（但保留表结构）
DELETE FROM Product;

-- TRUNCATE：更快地清空表（DDL，不可回滚）
TRUNCATE TABLE Product;
DELETE 是 DML，可以回滚；TRUNCATE 是 DDL，不能回滚但速度更快。

小结

  • SELECT + FROM + WHERE 是 SQL 最核心的查询模式
  • ORDER BY 排序，LIMIT/OFFSET 分页
  • INSERT 添加行，UPDATE 修改行，DELETE 删除行
  • WHERE 子句支持 AND/OR/NOT/IN/LIKE/IS NULL
  • 没有 WHERE 条件的 UPDATE/DELETE 会作用于所有行
