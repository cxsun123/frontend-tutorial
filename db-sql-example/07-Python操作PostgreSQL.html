07 - Python 操作 PostgreSQL

Python + PostgreSQL = CRUD

Python 的 psycopg2 库是操作 PostgreSQL 的事实标准。本章演示完整的增删改查和连接池管理。

安装与连接

安装 psycopg2 并建立连接

pip install psycopg2-binary
import psycopg2
from psycopg2 import Error

try:
    # 建立连接
    conn = psycopg2.connect(
        user="postgres",
        password="mysecret",
        host="127.0.0.1",
        port="5432",
        database="exampledb"
    )
    print("数据库连接成功！")

    # 获取游标
    cursor = conn.cursor()
    cursor.execute("SELECT VERSION()")
    version = cursor.fetchone()
    print(f"PostgreSQL 版本: {version[0]}")

except (Exception, Error) as err:
    print(f"连接错误: {err}")
finally:
    if conn:
        cursor.close()
        conn.close()

创建表

import psycopg2

conn = psycopg2.connect(
    database="exampledb", user="postgres",
    password="mysecret", host="127.0.0.1", port="5432"
)
cursor = conn.cursor()

create_table_sql = """
CREATE TABLE IF NOT EXISTS employees (
    id       SERIAL PRIMARY KEY,
    name     VARCHAR(100) NOT NULL,
    age      INTEGER,
    position VARCHAR(100),
    salary   NUMERIC(10, 2),
    hired_at DATE DEFAULT CURRENT_DATE
);
"""
cursor.execute(create_table_sql)
conn.commit()
print("表创建成功")

cursor.close()
conn.close()

插入数据

三种插入方式对比

# 方式一：参数分离（推荐，防止 SQL 注入）
sql = """INSERT INTO employees (name, age, position, salary)
         VALUES (%s, %s, %s, %s)"""
params = ('张三', 30, '软件工程师', 15000.00)
cursor.execute(sql, params)
conn.commit()
print(f"插入成功，影响 {cursor.rowcount} 行")

# 方式二：字典映射
sql = """INSERT INTO employees (name, age, position, salary)
         VALUES (%(name)s, %(age)s, %(pos)s, %(salary)s)"""
data = {'name': '李四', 'age': 28, 'pos': '产品经理', 'salary': 18000}
cursor.execute(sql, data)
conn.commit()

# 批量插入
employees = [
    ('王五', 35, '架构师', 25000),
    ('赵六', 26, '前端工程师', 13000),
    ('孙七', 32, 'DBA', 20000),
]
sql = """INSERT INTO employees (name, age, position, salary)
         VALUES (%s, %s, %s, %s)"""
cursor.executemany(sql, employees)
conn.commit()
print(f"批量插入 {cursor.rowcount} 行")
始终使用 %s 占位符传参，不要用 f-string 拼接 SQL，可防止 SQL 注入攻击。

查询数据

# 查询所有
cursor.execute("SELECT * FROM employees")
records = cursor.fetchall()
for row in records:
    print(f"ID={row[0]}, 姓名={row[1]}, 年龄={row[2]}, 职位={row[3]}, 薪资={row[4]}")

# 带条件的查询
cursor.execute(
    "SELECT * FROM employees WHERE salary > %s ORDER BY salary DESC",
    (15000,)
)
high_earners = cursor.fetchall()
print(f"高薪员工数: {len(high_earners)}")

# 获取单条
cursor.execute("SELECT * FROM employees WHERE id = %s", (1,))
emp = cursor.fetchone()
if emp:
    print(f"姓名: {emp[1]}, 职位: {emp[3]}")

# 查询一行一列
cursor.execute("SELECT COUNT(*) FROM employees")
count = cursor.fetchone()[0]
print(f"员工总数: {count}")

更新与删除

# 更新
update_sql = """
    UPDATE employees
    SET salary = %s, position = %s
    WHERE id = %s
"""
cursor.execute(update_sql, (20000, '高级工程师', 1))
conn.commit()
print(f"更新了 {cursor.rowcount} 行")

# 删除
delete_sql = "DELETE FROM employees WHERE id = %s"
cursor.execute(delete_sql, (5,))
conn.commit()
print(f"删除了 {cursor.rowcount} 行")

事务管理

try:
    # 开始事务（默认自动开始）
    cursor.execute("UPDATE accounts SET balance = balance - 1000 WHERE id = 1")
    cursor.execute("UPDATE accounts SET balance = balance + 1000 WHERE id = 2")
    # 提交
    conn.commit()
    print("转账成功")
except Exception as err:
    # 回滚
    conn.rollback()
    print(f"转账失败，已回滚: {err}")

连接池

使用 psycopg2.pool 管理数据库连接

from psycopg2 import pool

# 创建连接池（最小 2，最大 10 个连接）
connection_pool = pool.SimpleConnectionPool(
    2, 10,
    user="postgres",
    password="mysecret",
    host="127.0.0.1",
    port="5432",
    database="exampledb"
)

# 从池中获取连接
conn = connection_pool.getconn()
cursor = conn.cursor()
cursor.execute("SELECT * FROM employees")
rows = cursor.fetchall()

# 归还连接到池
cursor.close()
connection_pool.putconn(conn)

# 关闭所有连接
connection_pool.closeall()

ORM 框架简介

SQLAlchemy — Python 最流行的 ORM

# pip install sqlalchemy psycopg2-binary

from sqlalchemy import create_engine, Column, Integer, String, Numeric
from sqlalchemy.orm import declarative_base, Session

engine = create_engine(
    "postgresql://postgres:mysecret@localhost:5432/exampledb"
)
Base = declarative_base()

class Employee(Base):
    __tablename__ = 'employees'
    id = Column(Integer, primary_key=True)
    name = Column(String(100))
    age = Column(Integer)
    position = Column(String(100))
    salary = Column(Numeric(10, 2))

# 创建会话
session = Session(engine)

# 查询
emps = session.query(Employee).filter(Employee.salary > 15000).all()
for emp in emps:
    print(f"{emp.name} - {emp.position} - {emp.salary}")

session.close()
ORM 将数据库表映射为 Python 类，让开发者用面向对象的方式操作数据库，提高开发效率。

小结

  • psycopg2 是 Python 操作 PostgreSQL 的标准库
  • 传参使用 %s 占位符防止 SQL 注入
  • executemany() 实现批量插入
  • 使用事务确保操作的原子性
  • 连接池提高并发性能
  • ORM（如 SQLAlchemy）进一步提升开发效率
